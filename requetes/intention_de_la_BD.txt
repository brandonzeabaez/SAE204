

CREATE TABLE Aliment(
   idA NUMBER(6) PRIMARY KEY,
   nomAliment VARCHAR2(40)
);

CREATE TABLE Sauce(
   idS NUMBER(6) PRIMARY KEY,
   nomSauce VARCHAR2(40)
);

CREATE TABLE Ingredient(
   idI NUMBER(6) PRIMARY KEY,
   nomIngredient VARCHAR2(40)

);

CREATE TABLE Modele(
   idMo NUMBER(6) PRIMARY KEY,
   nomModele VARCHAR2(40)
);

CREATE TABLE Titre(
   idTi NUMBER(6) PRIMARY KEY,
   nomTitre CHAR(12),
   CONSTRAINT ck_nom_titre CHECK (nomTitre IN('PHILANTHROPE', 'PROTECTEUR', 'HONORABLE'))
);

CREATE TABLE Dignite(
   idD NUMBER(6) PRIMARY KEY,
   nomDignite CHAR(40),
   CONSTRAINT ck_dignite CHECK (nomDignite IN('MAITRE', 'GRAND CHANCELIER', 'GRAND MAITRE'))
);

CREATE TABLE Rang(
   idRa NUMBER(6) PRIMARY KEY,
   nomRang CHAR(9),
   CONSTRAINT ck_nomRang CHECK (nomRang IN('NOVICE', 'COMPAGNON'))
);

CREATE TABLE Legume(
   idL NUMBER(6),
   nomLegume VARCHAR2(40),
   PRIMARY KEY(idL)
);

CREATE TABLE Organisation(
   idO NUMBER(6),
   PRIMARY KEY(idO)
);

CREATE TABLE Organisme(
   numeroSiret CHAR(14),
   raisonSociale VARCHAR2(50),
   PRIMARY KEY(numeroSiret)
);

CREATE TABLE Grade(
   idG NUMBER(6),
   PRIMARY KEY(idG)
);

CREATE TABLE PasTresHautGradee(
   idG NUMBER(6),
   nomDeGrade CHAR(12),
   PRIMARY KEY(idG),
   FOREIGN KEY(idG) REFERENCES Grade(idG),
   CONSTRAINT ck_pas_haut_gradee CHECK (nomDeGrade IN('AFFILEE', 'SYMPATHISANT', 'ADHERENT'))
);

CREATE TABLE TresHautGradee(
   idG NUMBER(6),
   nomDeGrade CHAR(10),
   PRIMARY KEY(idG),
   FOREIGN KEY(idG) REFERENCES Grade(idG),
   CONSTRAINT ck_haut_gradee CHECK (nomDeGrade IN('CHEVALIER', 'DAME', 'GRAND CHEVALIER', 'HAUTE DAME', 'COMMANDEUR', 'GRAND CROIX'))
);

CREATE TABLE NomMachine(
   idNM VARCHAR2(50),
   nomMachine VARCHAR2(40),
   PRIMARY KEY(idNM)
);

CREATE TABLE TypeEntretien(
   idE NUMBER(6),
   typeEntretien VARCHAR2(40),
   PRIMARY KEY(idE)
);

CREATE TABLE Tenrac(
   idO NUMBER(6),
   numeroSiret CHAR(14),
   idT NUMBER(6),
   nomTenrac VARCHAR2(40),
   courrier VARCHAR2(50),
   telephone CHAR(13),
   adressePostal VARCHAR2(30),
   idG NUMBER(6) NOT NULL,
   idTi NUMBER(6),
   idD NUMBER(6),
   idRa NUMBER(6),
   PRIMARY KEY(idO, numeroSiret, idT),
   FOREIGN KEY(idO) REFERENCES Organisation(idO),
   FOREIGN KEY(numeroSiret) REFERENCES Organisme(numeroSiret),
   FOREIGN KEY(idG) REFERENCES Grade(idG),
   FOREIGN KEY(idTi) REFERENCES Titre(idTi),
   FOREIGN KEY(idD) REFERENCES Dignite(idD),
   FOREIGN KEY(idRa) REFERENCES Rang(idRa)
);

CREATE TABLE Ordre(
   idO NUMBER(6),
   nomOrdre VARCHAR2(40),
   territoire VARCHAR2(40),
   PRIMARY KEY(idO),
   FOREIGN KEY(idO) REFERENCES Organisation(idO)
);

CREATE TABLE Plat(
   idP NUMBER(6),
   nomPlat VARCHAR2(40),
   idL NUMBER(6),
   PRIMARY KEY(idP),
   FOREIGN KEY(idL) REFERENCES Legume(idL)
);

CREATE TABLE AdressePartenaire(
   idA NUMBER(6),
   adressePart VARCHAR2(120),
   idO NUMBER(6) NOT NULL,
   PRIMARY KEY(idA),
   FOREIGN KEY(idO) REFERENCES Ordre(idO)
);

CREATE TABLE Club(
   idO_1 NUMBER(6),
   nomClub VARCHAR2(40),
   adresseClub VARCHAR2(120),
   idO NUMBER(6) NOT NULL,
   PRIMARY KEY(idO_1),
   FOREIGN KEY(idO_1) REFERENCES Organisation(idO),
   FOREIGN KEY(idO) REFERENCES Ordre(idO)
);

CREATE TABLE Machine(
   idM NUMBER(6),
   idNM VARCHAR2(50) NOT NULL,
   idMo NUMBER(6) NOT NULL,
   PRIMARY KEY(idM),
   FOREIGN KEY(idNM) REFERENCES NomMachine(idNM),
   FOREIGN KEY(idMo) REFERENCES Modele(idMo)
);

CREATE TABLE Reunion(
   idReunion NUMBER(6),
   date_ DATE,
   nomReunion VARCHAR2(40),
   idO NUMBER(6) NOT NULL,
   numeroSiret CHAR(14) NOT NULL,
   idT NUMBER(6) NOT NULL,
   idA NUMBER(6) NOT NULL,
   PRIMARY KEY(idReunion, date_),
   FOREIGN KEY(idO, numeroSiret, idT) REFERENCES Tenrac(idO, numeroSiret, idT),
   FOREIGN KEY(idA) REFERENCES AdressePartenaire(idA)
);

CREATE TABLE Repas(
   idR NUMBER(6),
   intitule VARCHAR2(40),
   idM NUMBER(6) NOT NULL,
   PRIMARY KEY(idR),
   FOREIGN KEY(idM) REFERENCES Machine(idM)
);

CREATE TABLE Assiste(
   idO NUMBER(6),
   numeroSiret CHAR(14),
   idT NUMBER(6),
   idReunion NUMBER(6),
   date_ DATE,
   PRIMARY KEY(idO, numeroSiret, idT, idReunion, date_),
   FOREIGN KEY(idO, numeroSiret, idT) REFERENCES Tenrac(idO, numeroSiret, idT),
   FOREIGN KEY(idReunion, date_) REFERENCES Reunion(idReunion, date_)
);

CREATE TABLE Compose(
   idP NUMBER(6),
   idR NUMBER(6),
   PRIMARY KEY(idP, idR),
   FOREIGN KEY(idP) REFERENCES Plat(idP),
   FOREIGN KEY(idR) REFERENCES Repas(idR)
);

CREATE TABLE Constitue(
   idP NUMBER(6),
   idA NUMBER(6),
   PRIMARY KEY(idP, idA),
   FOREIGN KEY(idP) REFERENCES Plat(idP),
   FOREIGN KEY(idA) REFERENCES Aliment(idA)
);

CREATE TABLE Est_constitue(
   idS NUMBER(6),
   idI NUMBER(6),
   PRIMARY KEY(idS, idI),
   FOREIGN KEY(idS) REFERENCES Sauce(idS),
   FOREIGN KEY(idI) REFERENCES Ingredient(idI)
);

CREATE TABLE Contient(
   idP NUMBER(6),
   idS NUMBER(6),
   PRIMARY KEY(idP, idS),
   FOREIGN KEY(idP) REFERENCES Plat(idP),
   FOREIGN KEY(idS) REFERENCES Sauce(idS)
);

CREATE TABLE Organise(
   idReunion NUMBER(6),
   date_ DATE,
   idR NUMBER(6),
   PRIMARY KEY(idReunion, date_, idR),
   FOREIGN KEY(idReunion, date_) REFERENCES Reunion(idReunion, date_),
   FOREIGN KEY(idR) REFERENCES Repas(idR)
);

CREATE TABLE Evite(
   idO NUMBER(6),
   numeroSiret CHAR(14),
   idT NUMBER(6),
   idL NUMBER(6),
   motif VARCHAR2(50),
   PRIMARY KEY(idO, numeroSiret, idT, idL),
   FOREIGN KEY(idO, numeroSiret, idT) REFERENCES Tenrac(idO, numeroSiret, idT),
   FOREIGN KEY(idL) REFERENCES Legume(idL)
);

CREATE TABLE Necessite(
   idMo NUMBER(6),
   idNM VARCHAR2(50),
   idE NUMBER(6),
   periodicite VARCHAR2(30),
   PRIMARY KEY(idMo, idNM, idE),
   FOREIGN KEY(idMo) REFERENCES Modele(idMo),
   FOREIGN KEY(idNM) REFERENCES NomMachine(idNM),
   FOREIGN KEY(idE) REFERENCES TypeEntretien(idE)
);

CREATE TABLE Entretien(
   idO NUMBER(6),
   numeroSiret CHAR(14),
   idT NUMBER(6),
   idM NUMBER(6),
   idE NUMBER(6),
   date_ VARCHAR2(50),
   PRIMARY KEY(idO, numeroSiret, idT, idM, idE),
   FOREIGN KEY(idO, numeroSiret, idT) REFERENCES Tenrac(idO, numeroSiret, idT),
   FOREIGN KEY(idM) REFERENCES Machine(idM),
   FOREIGN KEY(idE) REFERENCES TypeEntretien(idE)
);
